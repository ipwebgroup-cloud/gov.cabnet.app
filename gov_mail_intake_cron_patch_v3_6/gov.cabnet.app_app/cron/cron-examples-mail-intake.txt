# gov.cabnet.app — Bolt Mail Intake cron examples
# Production-safe: imports Bolt Ride details emails into bolt_mail_intake only.
# Does not create EDXEIX jobs and does not submit anything live.

# Recommended during first production days: every 2 minutes
*/2 * * * * /usr/local/bin/php /home/cabnet/gov.cabnet.app_app/cli/import_bolt_mail.php --limit=250 --days=30 >> /home/cabnet/gov.cabnet.app_app/storage/logs/bolt_mail_intake.log 2>&1

# More conservative option: every 5 minutes
